def classify_age (age):
    if age['Age'] > 0 and age['Age'] <= 18 :
        return 'Child'
    elif age['Age'] > 18 and age['Age'] <= 60:
        return 'Adult'
    elif age['Age'] > 60:
        return 'Senior'
    
df['classify_age'] = df.apply(classify_age,axis=1)
